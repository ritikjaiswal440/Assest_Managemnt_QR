/**
 * Commits updates down to administrative log notes on the parent dashboard visibility layers.
 */
function handleParentRemarkRequest(payload) {
  const lock = LockService.getScriptLock();
  try {
    lock.waitLock(10000);
    
    if (payload.userRole.toLowerCase() !== 'admin') {
      return { success: false, message: "Security Violation: Privileged administrative level clearance authentication required.", data: null };
    }

    const ss = SpreadsheetApp.openById(MASTER_DB_ID);
    const pSheet = ss.getSheetByName('Parent_Tickets');
    const rawGrid = pSheet.getDataRange().getValues();
    const headers = rawGrid[0];
    
    const remarksColumnOffset = headers.indexOf('Admin_Remarks');
    if (remarksColumnOffset === -1) throw new Error("Schema Violation: Parameter column mapping field mismatch.");

    for(let i=1; i<rawGrid.length; i++) {
      if(rawGrid[i][0] === payload.parentId) {
        const cell = pSheet.getRange(i + 1, remarksColumnOffset + 1);
        const baselineData = cell.getValue() || '';
        const executionTime = Utilities.formatDate(new Date(), Session.getScriptTimeZone(), "MM/dd HH:mm");
        
        const appendedDataText = baselineData 
          ? baselineData + `\n[${executionTime} - Admin] ${payload.remark}` 
          : `[${executionTime} - Admin] ${payload.remark}`;
          
        cell.setValue(appendedDataText);
        SpreadsheetApp.flush(); // Force immediate spreadsheet update write-back to ensure immediate fetch visibility
        logSystemAction(payload.actorEmail, "PARENT_REMARK_APPENDED", `Updated audit remark parameters for ${payload.parentId}`);
        return { success: true, message: "Administrative metadata updates committed to master ticket tracker ledger layers.", data: null };
      }
    }
    throw new Error("Query Failure: Target structural data file entity key pointer missing execution reference map.");
  } catch(e) {
    return { success: false, message: e.toString(), data: null };
  } finally {
    lock.releaseLock();
  }
}

/**
 * Enables engineers to process localized log records updates, manage status updates, and execute closeout protocols on subtask rows.
 */
function handleChildUpdateRequest(payload) {
  const lock = LockService.getScriptLock();
  try {
    lock.waitLock(10000);
    const ss = SpreadsheetApp.openById(MASTER_DB_ID);
    const cSheet = ss.getSheetByName('Child_Tickets');
    const dataGrid = cSheet.getDataRange().getValues();
    const currentTimeStamp = new Date();
    
    const isAdmin = payload.userRole.toLowerCase() === 'admin';

    for(let i=1; i<dataGrid.length; i++) {
      if(dataGrid[i][0] === payload.childId) {
        const structuralRowOffset = i + 1;
        const rowRange = cSheet.getRange(structuralRowOffset, 1, 1, cSheet.getLastColumn());
        let currentRecordValues = rowRange.getValues()[0];

        // Security boundary confirmation: Check if resource matches target allocation pointer
        if(!isAdmin && currentRecordValues[3].toString().toLowerCase().trim() !== payload.actorEmail.toLowerCase().trim()) {
          return { success: false, message: "Security Violation: Data access restricted. Modification authorization bounds exceeded.", data: null };
        }

        if(payload.status === 'Closed' && currentRecordValues[5] !== 'Closed') {
          currentRecordValues[8] = currentTimeStamp; // Set Closed_Date column values
        }
        
        currentRecordValues[5] = payload.status; // Set operational status enum changes

        if (payload.remark.trim() !== '') {
          const humanTimestampMarker = Utilities.formatDate(currentTimeStamp, Session.getScriptTimeZone(), "MM/dd HH:mm");
          const actorLabel = isAdmin ? 'Admin' : 'Engineer';
          const originalRemarksBlock = currentRecordValues[6] || '';
          
          currentRecordValues[6] = originalRemarksBlock 
            ? originalRemarksBlock + `\n[${humanTimestampMarker} - ${actorLabel}] ${payload.remark}` 
            : `[${humanTimestampMarker} - ${actorLabel}] ${payload.remark}`;
        }

        rowRange.setValues([currentRecordValues]);
        evaluateAndEnforceParentStatus(payload.parentId, ss);
        SpreadsheetApp.flush(); // Force immediate spreadsheet write-back
        
        logSystemAction(payload.actorEmail, "CHILD_TICKET_UPDATED", `Modified parameters on child task identifier index row link ${payload.childId}`);
        return { success: true, message: "Worklog adjustments synchronized down to data service systems successfully.", data: null };
      }
    }
    throw new Error("Target operations context structure not identified.");
  } catch(e) {
    return { success: false, message: e.toString(), data: null };
  } finally {
    lock.releaseLock();
  }
}

/**
 * Tracks tickets by searching the Master Operations board first.
 * If not found, it falls back to the Client Intake Queue (Requests sheet).
 */
function handleTrackingRequest(payload) {
  const searchId = (payload.trackingId || '').trim();
  if (!searchId) return { success: false, message: "A tracking ID is required.", data: null };

  const lock = LockService.getScriptLock();
  try {
    lock.waitLock(10000);
    // Use MASTER_DB_ID if separated, otherwise getActiveSpreadsheet()
    const ss = SpreadsheetApp.openById(MASTER_DB_ID); 
    
    // Auto-migrate legacy scrambled rows
    migrateParentTickets(ss);
    
    let matchedTickets = []; 

    // --- 1. SEARCH MASTER TICKETS FIRST ---
    const pSheet = ss.getSheetByName('Parent_Tickets');
    if (pSheet) {
      const pData = pSheet.getDataRange().getValues();
      const pHeaders = pData[0];
      const parentIdIdx = pHeaders.indexOf('Parent_ID');
      const serviceReqIdx = pHeaders.indexOf('Service_Request_ID');

      for (let i = 1; i < pData.length; i++) {
        if (pData[i][parentIdIdx] === searchId || pData[i][serviceReqIdx] === searchId) {
          matchedTickets.push({
            parentId: pData[i][parentIdIdx],
            serviceRequestId: pData[i][serviceReqIdx],
            company: pData[i][pHeaders.indexOf('Company_Name')],
            location: pData[i][pHeaders.indexOf('Location')],
            category: pData[i][pHeaders.indexOf('Category')],
            issueType: pData[i][pHeaders.indexOf('Issue_Type')],
            description: pData[i][pHeaders.indexOf('Description')],
            status: pData[i][pHeaders.indexOf('Status')],
            openDate: pData[i][pHeaders.indexOf('Open_Date')],
            closeDate: pData[i][pHeaders.indexOf('Close_Date')],
            productMake: pHeaders.indexOf('ProductMake') !== -1 ? pData[i][pHeaders.indexOf('ProductMake')] : (pHeaders.indexOf('Brand') !== -1 ? pData[i][pHeaders.indexOf('Brand')] : ''),
            productModel: pHeaders.indexOf('ProductModel') !== -1 ? pData[i][pHeaders.indexOf('ProductModel')] : (pHeaders.indexOf('Model') !== -1 ? pData[i][pHeaders.indexOf('Model')] : ''),
            productSerial: pHeaders.indexOf('ProductSerial') !== -1 ? pData[i][pHeaders.indexOf('ProductSerial')] : (pHeaders.indexOf('Serial') !== -1 ? pData[i][pHeaders.indexOf('Serial')] : '')
          });
        }
      }
    }

    // --- 2. IF NOT FOUND, SEARCH INTAKE QUEUE ---
    if (matchedTickets.length === 0) {
      const rSheet = ss.getSheetByName('Requests');
      const itemSheet = ss.getSheetByName('Request_Items');
      if (rSheet) {
        const rData = rSheet.getDataRange().getValues();
        let itemsData = [];
        let itemHeaders = [];
        if (itemSheet) {
          itemsData = itemSheet.getDataRange().getValues();
          if (itemsData.length > 0) {
            itemHeaders = itemsData[0];
          }
        }
        
        const reqIdIdx = itemHeaders.indexOf('RequestID');
        const makeIdx = itemHeaders.indexOf('ProductMake') !== -1 ? itemHeaders.indexOf('ProductMake') : itemHeaders.indexOf('ProductBrand');
        const modelIdx = itemHeaders.indexOf('ProductModel');
        const serialIdx = itemHeaders.indexOf('ProductSerial');

        for (let i = 1; i < rData.length; i++) {
          if (rData[i][0] === searchId) { // RequestID is column A
            let products = [];
            if (itemSheet && itemsData.length > 1 && reqIdIdx !== -1) {
              for (let k = 1; k < itemsData.length; k++) {
                if (String(itemsData[k][reqIdIdx]) === String(rData[i][0])) {
                  products.push({
                    productMake: makeIdx !== -1 ? itemsData[k][makeIdx] : '',
                    productModel: modelIdx !== -1 ? itemsData[k][modelIdx] : '',
                    productSerial: serialIdx !== -1 ? itemsData[k][serialIdx] : ''
                  });
                }
              }
            }

            let make = '';
            let model = '';
            let serial = '';
            if (products.length > 0) {
              make = products.map(p => p.productMake).filter(Boolean).join(', ');
              model = products.map(p => p.productModel).filter(Boolean).join(', ');
              serial = products.map(p => p.productSerial).filter(Boolean).join(', ');
            }

            matchedTickets.push({
              parentId: "Pending Assignment", // No Master ID yet
              serviceRequestId: rData[i][0],
              company: rData[i][3], // CompanyName
              location: rData[i][4], // Location
              category: rData[i][7], // Category
              issueType: "Pending Triage",
              description: rData[i][8], // IssueDescription
              status: "Received", // Custom queue status
              openDate: rData[i][1], // Timestamp
              closeDate: null,
              productMake: make || 'N/A',
              productModel: model || 'N/A',
              productSerial: serial || 'N/A'
            });
          }
        }
      }
    }

    // --- 3. RETURN RESULTS ---
    if (matchedTickets.length === 0) {
      return { success: false, message: "No matching support tickets found. Please check your ID and try again.", data: null };
    }

    return { success: true, message: "Tickets located.", data: { tickets: matchedTickets } };

  } catch (e) {
    return { success: false, message: "Database Error: " + e.toString(), data: null };
  } finally {
    lock.releaseLock();
  }
}

/**
 * Executes the final closeout protocol for Master Parent Tickets.
 * Uses dynamic header indexing to ensure the correct columns are always updated.
 */
function closeParentTicket(parentId) {
  const lock = LockService.getScriptLock();
  try {
    lock.waitLock(10000);
    const ss = SpreadsheetApp.openById(MASTER_DB_ID);

    // 1. Validation: Ensure all child tasks are closed first
    const childCollectionObjects = fetchSheetRecordsAsObjects('Child_Tickets');
    const activeChildren = childCollectionObjects.filter(c => c.Parent_ID === parentId && c.Status !== 'Closed');
    
    if (activeChildren.length > 0) {
      throw new Error(`Cannot close Parent Ticket. There are ${activeChildren.length} active child sub-tasks remaining.`);
    }

    const pSheet = ss.getSheetByName('Parent_Tickets');
    const pData = pSheet.getDataRange().getValues();
    const headers = pData[0]; // Grab the top row to map names dynamically
    const date = new Date();

    // Map the exact column locations
    const parentIdIdx = headers.indexOf('Parent_ID');
    const serviceReqIdx = headers.indexOf('Service_Request_ID');
    const statusIdx = headers.indexOf('Status');
    const closeDateIdx = headers.indexOf('Close_Date');
    const resolvedDaysIdx = headers.indexOf('Resolved_Days');
    const openDateIdx = headers.indexOf('Open_Date');
    const companyIdx = headers.indexOf('Company_Name');
    const reqByIdx = headers.indexOf('Requested_By');
    const brandIdx = headers.indexOf('ProductMake') !== -1 ? headers.indexOf('ProductMake') : headers.indexOf('Brand');
    const modelIdx = headers.indexOf('ProductModel') !== -1 ? headers.indexOf('ProductModel') : headers.indexOf('Model');
    const serialIdx = headers.indexOf('ProductSerial') !== -1 ? headers.indexOf('ProductSerial') : headers.indexOf('Serial');
    const descIdx = headers.indexOf('Description');
    const emailIdx = headers.indexOf('Client_Email');
    const phoneIdx = headers.indexOf('PhoneNumber');
    const roomIdx = headers.indexOf('Room_Name');
    const locationIdx = headers.indexOf('Location');

    for (let i = 1; i < pData.length; i++) {
      if (pData[i][parentIdIdx] === parentId) {
        const rowRange = pSheet.getRange(i + 1, 1, 1, pSheet.getLastColumn());
        let row = rowRange.getValues()[0];

        // Pull details for the email using dynamic indices
        const companyName = companyIdx > -1 ? row[companyIdx] : 'Client'; 
        const reqBy = reqByIdx > -1 ? row[reqByIdx] : 'N/A'; 
        const brand = brandIdx > -1 ? row[brandIdx] : 'N/A'; 
        const model = modelIdx > -1 ? row[modelIdx] : 'N/A'; 
        const serial = serialIdx > -1 ? row[serialIdx] : 'N/A'; 
        const description = descIdx > -1 ? row[descIdx] : ''; 
        const phoneNumber = phoneIdx > -1 ? (row[phoneIdx] || 'NA') : 'NA';
        const serviceRequestId = serviceReqIdx > -1 ? row[serviceReqIdx] : 'N/A';
        const roomName = roomIdx > -1 ? (row[roomIdx] || 'NA') : 'NA';
        const location = locationIdx > -1 ? (row[locationIdx] || 'NA') : 'NA';
        
        const openDate = (openDateIdx > -1 && row[openDateIdx]) ? new Date(row[openDateIdx]) : date; 
        
        const formattedOpenDate = Utilities.formatDate(openDate, Session.getScriptTimeZone(), "MMMM dd, yyyy 'at' hh:mm a");
        const formattedCloseDate = Utilities.formatDate(date, Session.getScriptTimeZone(), "MMMM dd, yyyy 'at' hh:mm a");
        
        let days = Math.ceil((date - openDate) / (1000 * 60 * 60 * 24));
        if (days < 1) days = 1;

        // TARGETED UPDATES: Inject new data into the exact mapped columns
        if (closeDateIdx > -1) row[closeDateIdx] = date;
        if (resolvedDaysIdx > -1) row[resolvedDaysIdx] = days;
        if (statusIdx > -1) row[statusIdx] = 'Closed';
        
        const clientEmail = emailIdx > -1 ? row[emailIdx] : ''; 
        
        // Write the targeted data back to the sheet
        rowRange.setValues([row]);
        SpreadsheetApp.flush(); // Force immediate spreadsheet write-back
        
        // Log the action securely
        logSystemAction("SYSTEM", "PARENT_TICKET_CLOSED", `Finalized master ticket ${parentId} in ${days} days.`);

        let mailStatus = "";
        
        // 2. Dispatch the automated closure email to the client
        if (clientEmail && clientEmail.toString().trim() !== '') {
          try {
            const messageBody = `
              <p>Hello <b>${companyName}</b>,</p>
              <p>We are pleased to inform you that your support ticket has been successfully resolved and is now closed.</p>
              <div style="background-color: #f9f9f9; border-left: 4px solid #27ae60; padding: 15px; margin: 20px 0;">
                <p style="margin: 0 0 5px 0;"><strong>Ticket ID:</strong> ${parentId}</p>
                <p style="margin: 0 0 5px 0;"><strong>Service Request ID (SR):</strong> ${serviceRequestId || 'N/A'}</p>
                <p style="margin: 0 0 5px 0;"><strong>Opened On:</strong> ${formattedOpenDate}</p>
                <p style="margin: 0 0 15px 0;"><strong>Closed On:</strong> ${formattedCloseDate}</p>
                <p style="margin: 0 0 5px 0;"><strong>Requested By:</strong> ${reqBy}</p>
                <p style="margin: 0 0 5px 0;"><strong>Site / Room:</strong> ${location} | Room: ${roomName}</p>
                <p style="margin: 0 0 5px 0;"><strong>Phone Number:</strong> ${phoneNumber}</p>
                ${(brand === 'N/A' && model === 'N/A' && serial === 'N/A') ? '<p style="margin: 0 0 5px 0;"><strong>Hardware Details:</strong> NA</p>' : `<h4 style="margin: 15px 0 5px; font-size: 13px; color: #64748b; text-transform: uppercase;">Hardware Details</h4><table style="width: 100%; border-collapse: collapse; font-size: 13px; margin-bottom: 15px; border: 1px solid #e2e8f0;"><thead><tr style="background-color: #f1f5f9; text-align: left;"><th style="padding: 8px; border: 1px solid #e2e8f0;">Make</th><th style="padding: 8px; border: 1px solid #e2e8f0;">Model</th><th style="padding: 8px; border: 1px solid #e2e8f0;">S/N</th></tr></thead><tbody><tr><td style="padding: 8px; border: 1px solid #e2e8f0;">${brand === 'N/A' ? 'NA' : brand}</td><td style="padding: 8px; border: 1px solid #e2e8f0;">${model === 'N/A' ? 'NA' : model}</td><td style="padding: 8px; border: 1px solid #e2e8f0;">${serial === 'N/A' ? 'NA' : serial}</td></tr></tbody></table>`}
                <p style="margin: 0 0 5px 0; color: #34495e;"><strong>Client Issue:</strong><br>${description}</p>
                <hr style="border: 0; border-top: 1px solid #ddd; margin: 15px 0;">
                <p style="margin: 0 0 5px 0;"><strong>Status:</strong> <span style="color: #27ae60; font-weight:bold;">Closed</span></p>
                <p style="margin: 0;"><strong>Total Resolution Time:</strong> ${days} Day(s)</p>
              </div>
              <p>You can view the full history by tracking your ticket on the AVD Pro-Support Portal.</p>
            `;
            MailApp.sendEmail({
              to: clientEmail.toString().trim(),
              subject: `[AVD ProSupport] Support Ticket Resolved: ${parentId} [SR Ref: ${serviceRequestId || 'N/A'}] [${companyName}]`,
              htmlBody: generateBrandedHtmlEmailTemplate("Support Ticket Resolved", messageBody)
            });
          } catch(emailErr) {
            mailStatus = " (Warning: Client email failed to send.)";
          }
        }

        return { success: true, message: 'Master ticket successfully closed.' + mailStatus, data: null };
      }
    }
    throw new Error('Target Parent Ticket not found in database.');
  } catch (e) {
    return { success: false, message: e.toString(), data: null };
  } finally {
    lock.releaseLock();
  }
}

/**
 * Resolves state management flows dynamically based on subtask row configuration matrix structures.
 */
function evaluateAndEnforceParentStatus(parentId, ss) {
  const childCollectionObjects = fetchSheetRecordsAsObjects('Child_Tickets');
  const contextIncidentTasks = childCollectionObjects.filter(c => c.Parent_ID === parentId);
  
  let targetDerivedStatus = 'Opened';
  if (contextIncidentTasks.length > 0) {
    const contextEvaluationStatusMatrix = contextIncidentTasks.every(c => c.Status === 'Closed');
    targetDerivedStatus = contextEvaluationStatusMatrix ? 'Ready to Close' : 'In Progress';
  }

  const pSheet = ss.getSheetByName('Parent_Tickets');
  const dataMatrix = pSheet.getDataRange().getValues();
  
  for(let i=1; i<dataMatrix.length; i++) {
    // Prevent overriding finalized, closed tickets automatically
    if(dataMatrix[i][0] === parentId && dataMatrix[i][19] !== 'Closed') {
      pSheet.getRange(i + 1, 20).setValue(targetDerivedStatus); // Column 20 corresponds to Status
      break;
    }
  }
}

/**
 * Searches the 'Users' sheet to map an engineer's name to their registered email.
 */
function getEngineerEmailByName(engineerName) {
  if (!engineerName) return "";
  
  try {
    const ss = SpreadsheetApp.openById(MASTER_DB_ID);
    const userSheet = ss.getSheetByName('Users');
    if (!userSheet) return "";
    
    const data = userSheet.getDataRange().getValues();
    if (data.length <= 1) return "";
    
    // Find column positions dynamically
    const headers = data[0].map(h => h.toString().trim().toLowerCase());
    const nameIndex = headers.indexOf('name');
    const emailIndex = headers.indexOf('email');
    const roleIndex = headers.indexOf('role');
    
    if (nameIndex === -1 || emailIndex === -1) return "";
    
    // Attempt exact match first
    for (let i = 1; i < data.length; i++) {
      const name = String(data[i][nameIndex]).trim();
      const email = String(data[i][emailIndex]).trim();
      const role = roleIndex !== -1 ? String(data[i][roleIndex]).trim().toLowerCase() : "";
      
      if (name.toLowerCase() === engineerName.trim().toLowerCase() && role.includes('engineer')) {
        return email;
      }
    }
    
    // Fallback to substring matching
    for (let i = 1; i < data.length; i++) {
      const name = String(data[i][nameIndex]).trim();
      const email = String(data[i][emailIndex]).trim();
      const role = roleIndex !== -1 ? String(data[i][roleIndex]).trim().toLowerCase() : "";
      
      if (name.toLowerCase().includes(engineerName.trim().toLowerCase()) && role.includes('engineer')) {
        return email;
      }
    }
  } catch (e) {
    Logger.log("Error in getEngineerEmailByName: " + e.toString());
  }
  
  return "";
}

/**
 * Queues a status update request email to be sent asynchronously, or sends it synchronously on permission failure.
 */
function handlePingEngineerRequest(payload) {
  try {
    const parentId = payload.parentId;
    const engineerName = payload.engineerName;

    if (!engineerName || !parentId) {
      throw new Error("Missing mandatory parameters: engineerName or parentId.");
    }

    try {
      // 1. Attempt programmatic trigger to dispatch asynchronously
      const trigger = ScriptApp.newTrigger('asyncSendPingEmail')
        .timeBased()
        .after(1000)
        .create();

      const triggerId = trigger.getUniqueId();

      // 2. Cache the payload associated with the unique trigger ID
      const cacheKey = 'ping_payload_' + triggerId;
      CacheService.getScriptCache().put(cacheKey, JSON.stringify(payload), 300);

      return {
        success: true,
        message: "Status update request received. Dispatching notification in background.",
        data: null
      };

    } catch (triggerError) {
      // Fallback: If trigger permissions are denied, send the email synchronously immediately
      Logger.log("Trigger creation failed. Falling back to synchronous dispatch: " + triggerError.toString());

      const childId = payload.childId || 'N/A';
      const clientCompany = payload.clientCompany || 'N/A';
      const location = payload.location || 'N/A';
      const issue = payload.issue || 'N/A';
      const actorEmail = payload.actorEmail || 'Admin';
      const phoneNumber = payload.phoneNumber || 'N/A';
      const roomName = payload.roomName || 'N/A';

      let engineerEmail = getEngineerEmailByName(engineerName);
      if (!engineerEmail && payload.engineerEmail) {
        engineerEmail = payload.engineerEmail;
      }

      if (!engineerEmail) {
        throw new Error("Could not resolve email address for engineer: " + engineerName);
      }

      const subject = `[AVD ProSupport] Ticket Assigned: ${parentId}`;
      
      const statusUpdateMarkup = `
        <p>Hello <b>${engineerName}</b>,</p>
        <p>An administrator is requesting an immediate status update on the following operational task assigned to you.</p>
        <div style="background-color: #f9f9f9; border-left: 4px solid #1abc9c; padding: 15px; margin: 20px 0;">
          <p style="margin: 0 0 5px 0;"><strong>Master Ticket ID Link:</strong> ${parentId}</p>
          <p style="margin: 0 0 5px 0;"><strong>Your Target Operation ID:</strong> ${childId}</p>
          <p style="margin: 0 0 5px 0;"><strong>Client Location:</strong> ${clientCompany} [${location} | Room: ${roomName}]</p>
          <p style="margin: 0 0 5px 0;"><strong>Phone Number:</strong> ${phoneNumber}</p>
          <p style="margin: 0 0 15px 0; color: #34495e;"><strong>Incident Context / Issue:</strong><br>${issue}</p>
          <hr style="border: 0; border-top: 1px solid #ddd; margin: 15px 0;">
          <p style="margin: 0; color: #e67e22;"><strong>Action Item:</strong><br>Log into the ProSupport Portal to update your task work logs and remarks.</p>
        </div>
      `;

      const htmlOutput = generateBrandedHtmlEmailTemplate("Status Update Request", statusUpdateMarkup);
      const searchQuery = `subject:"${subject}" to:${engineerEmail}`;
      const threads = GmailApp.search(searchQuery, 0, 1);
      
      if (threads.length > 0) {
        threads[0].replyAll("Please view this email in an HTML-compatible client.", {
          htmlBody: htmlOutput
        });
      } else {
        MailApp.sendEmail({
          to: engineerEmail,
          subject: subject,
          htmlBody: htmlOutput,
          name: "AV Dynamic Operations"
        });
      }

      // Audit log the action
      logSystemAction(actorEmail, "ENGINEER_PINGED_FOR_UPDATE", `Emailed status update request to engineer ${engineerName} (${engineerEmail}) for task ${childId} linked to ticket ${parentId} (Synchronous Fallback)`);

      return {
        success: true,
        message: "Status update request email dispatched successfully (via synchronous fallback).",
        data: null
      };
    }

  } catch (error) {
    return {
      success: false,
      message: "Notification Exception: " + error.toString(),
      data: null
    };
  }
}

/**

 * Automatically cleans up the trigger and cache value on completion.

    if (!triggerId) {
      Logger.log("Trigger execution context did not pass triggerUid.");
      return;
    }

    const cacheKey = 'ping_payload_' + triggerId;
    const cacheVal = CacheService.getScriptCache().get(cacheKey);
    if (!cacheVal) {
      Logger.log("No cached payload found for trigger: " + triggerId);
      return;
    }

    const payload = JSON.parse(cacheVal);
    const parentId = payload.parentId;
    const childId = payload.childId || 'N/A';
    const engineerName = payload.engineerName;
    const clientCompany = payload.clientCompany || 'N/A';
    const location = payload.location || 'N/A';

    const actorEmail = payload.actorEmail || 'Admin';
    const phoneNumber = payload.phoneNumber || 'N/A';

          <p style="margin: 0 0 5px 0;"><strong>Client Location:</strong> ${clientCompany} [${location} | Room: ${roomName}]</p>

          <p style="margin: 0 0 15px 0; color: #34495e;"><strong>Incident Context / Issue:</strong><br>${issue}</p>
          <hr style="border: 0; border-top: 1px solid #ddd; margin: 15px 0;">

      logSystemAction(actorEmail, "ENGINEER_PINGED_FOR_UPDATE", `Emailed status update request to engineer ${engineerName} (${engineerEmail}) for task ${childId} linked to ticket ${parentId}`);

  } catch (error) {
    Logger.log("Error in asyncSendPingEmail execution: " + error.toString());
  } finally {
    // Programmatically clean up the one-time trigger so it doesn't leak
    if (triggerId) {
      deleteTriggerByUid(triggerId);
    }
  }
}

/**
 * Removes a programmatic trigger by its unique ID.
 */
function deleteTriggerByUid(triggerId) {
  if (!triggerId) return;
  try {
    const triggers = ScriptApp.getProjectTriggers();
    for (let i = 0; i < triggers.length; i++) {
      if (triggers[i].getUniqueId() === triggerId) {
        ScriptApp.deleteTrigger(triggers[i]);
        break;
      }
    }
  } catch (e) {
    Logger.log("Error deleting trigger " + triggerId + ": " + e.toString());
  }
}
