function handleValidateRef(ref) {
  if (!ref) return { success: false, message: 'No reference provided.', data: null };
  
  const ss = SpreadsheetApp.openById(MASTER_DB_ID);
  const sheet = ss.getSheetByName(PORTAL_CONFIG.SHEET_MAPPING);
  if (!sheet) return { success: false, message: 'Mapping database not found.', data: null };

  const data = sheet.getDataRange().getValues();
  for (let i = 1; i < data.length; i++) {
    if (data[i][0] === ref) {
      if (data[i][2] === true || data[i][2] === 'TRUE' || data[i][2] === 'true') {
        return { success: true, message: 'Authorized', data: { companyName: data[i][1] } };
      }
      return { success: false, message: 'This client link is inactive.', data: null };
    }
  }
  return { success: false, message: 'Invalid reference code.', data: null };
}

function processClientRequest(payload) {
  const lock = LockService.getScriptLock();
  lock.waitLock(10000); 
  
  try {
    const ss = SpreadsheetApp.openById(MASTER_DB_ID);
    
    // Self-heal and standardize headers dynamically
    standardizeSpreadsheetHeaders(ss);

    const reqSheet = ss.getSheetByName(PORTAL_CONFIG.SHEET_REQUESTS);
    const itemSheet = ss.getSheetByName(PORTAL_CONFIG.SHEET_ITEMS);
    const mapSheet = ss.getSheetByName(PORTAL_CONFIG.SHEET_MAPPING);
    
    if (!reqSheet || !itemSheet) throw new Error("Client database sheets are missing.");

    // 1. Verify Client
    let companyName = "Unknown Client";
    const mapData = mapSheet.getDataRange().getValues();
    let isValid = false;
    for (let i = 1; i < mapData.length; i++) {
      if (mapData[i][0] === payload.ref && (mapData[i][2] === true || mapData[i][2] === 'TRUE' || mapData[i][2] === 'true')) {
        isValid = true;
        companyName = mapData[i][1];
        break;
      }
    }
    if (!isValid) throw new Error("Invalid reference code.");
    
    const timestamp = new Date();
    const requestId = generateRequestId(reqSheet, timestamp);
    
    // 2. Process File Upload
    let invoiceUrl = "";
    let attachmentBlob = null;
    if (payload.fileData && payload.fileData.base64) {
      const folder = DriveApp.getFolderById(PORTAL_CONFIG.DRIVE_FOLDER_ID);
      const bytes = Utilities.base64Decode(payload.fileData.base64);
      const blob = Utilities.newBlob(bytes, payload.fileData.mimeType, payload.fileData.filename);
      const tsString = Utilities.formatDate(timestamp, Session.getScriptTimeZone(), "yyyyMMdd_HHmmss");
      const safeName = payload.fileData.filename.replace(/[^a-zA-Z0-9.\-_]/g, '_');
      blob.setName(`${requestId}_${tsString}_${safeName}`);
      invoiceUrl = folder.createFile(blob).getUrl();
      attachmentBlob = blob;
    }
    
    // 3. Save to Client Requests Sheet (Intake Queue)
    reqSheet.appendRow([
      requestId, timestamp, sanitize(payload.email), companyName,
      sanitize(payload.location), sanitize(payload.requesterName),
      sanitize(payload.roomName), sanitize(payload.category),
      sanitize(payload.issueDescription), invoiceUrl, sanitize(payload.phoneNumber)
    ]);

    // 4. Save to Client Request Items Sheet
    if (payload.products && payload.products.length > 0) {
      const itemsToSave = payload.products.map((prod, index) => [
        requestId, 
        index + 1, 
        sanitize(prod.productMake || prod.brand || ''), 
        sanitize(prod.productModel || prod.model || ''), 
        sanitize(prod.productSerial || prod.serial || '')
      ]);
      itemSheet.getRange(itemSheet.getLastRow() + 1, 1, itemsToSave.length, itemsToSave[0].length).setValues(itemsToSave);
    }
    
    // 5. Send Notification Emails
    sendNotifications(requestId, payload, companyName, invoiceUrl, attachmentBlob);

    return { success: true, message: "Ticket logged successfully", data: { requestId: requestId } };

  } catch (err) {
    return { success: false, message: err.message, data: null };
  } finally {
    lock.releaseLock();
  }
}

function generateRequestId(reqSheet, dateObj) {
  const dateStr = Utilities.formatDate(dateObj, Session.getScriptTimeZone(), "yyyyMMdd");
  const prefix = `SR-${dateStr}-`;
  const lastRow = reqSheet.getLastRow();
  let counter = 1;
  if (lastRow > 1) {
    const lastId = reqSheet.getRange(lastRow, 1).getValue().toString();
    if (lastId.startsWith(prefix)) {
      const parts = lastId.split('-');
      if (parts.length === 3) {
        counter = parseInt(parts[2], 10) + 1;
      }
    }
  }
  return `${prefix}${counter.toString().padStart(4, '0')}`;
}

function sanitize(str) {
  if (!str) return '';
  return str.toString().replace(/^[=\+\-@]/, "'$&");
}

/**
 * Marks an intake queue service request as manually archived/fully processed.
 * Standardizes the 'Archived' column and sets it to 'TRUE' for the matching row.
 */
function handleArchiveRequest(payload) {
  const requestId = (payload.requestId || '').trim();
  if (!requestId) return { success: false, message: "A request ID is required.", data: null };

  const lock = LockService.getScriptLock();
  try {
    lock.waitLock(10000);
    const ss = SpreadsheetApp.openById(MASTER_DB_ID);
    const rSheet = ss.getSheetByName(PORTAL_CONFIG.SHEET_REQUESTS);
    if (!rSheet) throw new Error("Requests sheet not found.");

    const rData = rSheet.getDataRange().getValues();
    const rHeaders = rData[0];
    let archivedIdx = rHeaders.indexOf('Archived');
    
    // If Archived header doesn't exist, append it dynamically
    if (archivedIdx === -1) {
      archivedIdx = rHeaders.length;
      rSheet.getRange(1, archivedIdx + 1).setValue('Archived');
      rSheet.getRange(1, archivedIdx + 1).setFontWeight('bold');
    }

    let found = false;
    for (let i = 1; i < rData.length; i++) {
      if (String(rData[i][0]) === requestId) {
        rSheet.getRange(i + 1, archivedIdx + 1).setValue('TRUE');
        found = true;
        break;
      }
    }

    if (!found) {
      return { success: false, message: "Request ID " + requestId + " not found in database.", data: null };
    }

    SpreadsheetApp.flush();
    return { success: true, message: "Request " + requestId + " archived successfully.", data: null };

  } catch (e) {
    return { success: false, message: "Error archiving request: " + e.toString(), data: null };
  } finally {
    lock.releaseLock();
  }
}

/**
 * Unmarks an intake queue service request as archived, returning it to the active queue.
 * Sets the 'Archived' column to 'FALSE' for the matching row.
 */
function handleUnarchiveRequest(payload) {
  const requestId = (payload.requestId || '').trim();
  if (!requestId) return { success: false, message: "A request ID is required.", data: null };

  const lock = LockService.getScriptLock();
  try {
    lock.waitLock(10000);
    const ss = SpreadsheetApp.openById(MASTER_DB_ID);
    const rSheet = ss.getSheetByName(PORTAL_CONFIG.SHEET_REQUESTS);
    if (!rSheet) throw new Error("Requests sheet not found.");

    const rData = rSheet.getDataRange().getValues();
    const rHeaders = rData[0];
    let archivedIdx = rHeaders.indexOf('Archived');
    
    if (archivedIdx === -1) {
      archivedIdx = rHeaders.length;
      rSheet.getRange(1, archivedIdx + 1).setValue('Archived');
      rSheet.getRange(1, archivedIdx + 1).setFontWeight('bold');
    }

    let found = false;
    for (let i = 1; i < rData.length; i++) {
      if (String(rData[i][0]) === requestId) {
        rSheet.getRange(i + 1, archivedIdx + 1).setValue('FALSE');
        found = true;
        break;
      }
    }

    if (!found) {
      return { success: false, message: "Request ID " + requestId + " not found in database.", data: null };
    }

    SpreadsheetApp.flush();
    return { success: true, message: "Request " + requestId + " unarchived successfully.", data: null };

  } catch (e) {
    return { success: false, message: "Error unarchiving request: " + e.toString(), data: null };
  } finally {
    lock.releaseLock();
  }
}
