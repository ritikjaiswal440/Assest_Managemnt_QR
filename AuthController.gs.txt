 * The active router uses handleAuthenticationRequest(payload) from AuthService.gs.txt.
 */

const DB_ID = '1iyrvEQ_El3rmnCOccXHrTaKkuAQVLgvln3RG8_eoPjs'; // Store the ID securely

function handleLogin(payload) {
  const email = payload.email;
  const password = payload.password; // Note: In a true prod environment, hash passwords
  
  if (!email || !password) {
    return { success: false, message: "Email and password are required.", data: null };
  }

  const ss = SpreadsheetApp.openById(DB_ID);
  const userSheet = ss.getSheetByName('Users');
  
  if (!userSheet) {
    return { success: false, message: "Database configuration error.", data: null };
  }

  const data = userSheet.getDataRange().getValues();
  const headers = data[0]; // [Name, Email, Role, Password, Company_Name, Projects, Status...]
  
  // Find column indexes dynamically to prevent breaking if columns are moved
  const emailIdx = headers.indexOf('Email');
  const passIdx = headers.indexOf('Password');
  const statusIdx = headers.indexOf('Status');
  
  // Search for the user
  for (let i = 1; i < data.length; i++) {
    let row = data[i];
    
    if (row[emailIdx].toString().toLowerCase() === email.toLowerCase()) {
      if (row[passIdx].toString() === password) {
        
        if (row[statusIdx] !== 'Active') {
          return { success: false, message: "Account is disabled or pending.", data: null };
        }

        // Parse associated projects/locations. 
        // A single client might oversee multiple project sites, stored as a comma-separated list.
        const projectListRaw = row[headers.indexOf('Projects')] || "";
        const associatedProjects = projectListRaw.split(',').map(p => p.trim()).filter(p => p !== "");
        
        // Construct the safe user object to return to React (never return the password)
        const userProfile = {
          name: row[headers.indexOf('Name')],
          email: row[emailIdx],
          role: row[headers.indexOf('Role')],
          company: row[headers.indexOf('Company_Name')],
          projects: associatedProjects, 
          lastLogin: new Date().toISOString()
        };
        
        // Update Last_Login timestamp in the sheet
        userSheet.getRange(i + 1, headers.indexOf('Last_Login') + 1).setValue(userProfile.lastLogin);
        
        return { 
          success: true, 
          message: `Welcome back, ${userProfile.name}`, 
          data: { user: userProfile } 
        };
      } else {
        return { success: false, message: "Invalid password.", data: null };
      }
    }
  }
  
  return { success: false, message: "User not found.", data: null };
}
