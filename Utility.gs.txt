/**
 * Transforms standard 2D sheet matrices rows into JSON array objects mapping columns to camelCase header strings.
 */
function fetchSheetRecordsAsObjects(sheetName) {
  const ss = SpreadsheetApp.openById(MASTER_DB_ID);
  const targetSheet = ss.getSheetByName(sheetName);
  if (!targetSheet) return [];
  
  const rawDataGrid = targetSheet.getDataRange().getDisplayValues();
  if (rawDataGrid.length < 2) return [];
  
  const headers = rawDataGrid.shift();
  return rawDataGrid.map(row => {
    let mappingEntityInstance = {};
    headers.forEach((h, indexPointer) => {
      // Formats header strings into clean key variables (e.g., 'Parent_ID' becomes 'Parent_ID')
      const operationalCleanKey = h.trim().replace(/\s+/g, '_');
      mappingEntityInstance[operationalCleanKey] = row[indexPointer];
    });
    return mappingEntityInstance;
  });
}

/**
 * Calculates standardized tracking alphanumeric indices matching fiscal year boundaries (e.g., AVD/PT/26-27/0004).
 */
function calculateIncrementalIdentifier(sheet, sequencePrefix) {
  const calendarDate = new Date();
  let financialYearStartMarker = calendarDate.getFullYear();
  
  // Fiscal target alignment calculation shifting on April 1st
  if (calendarDate.getMonth() < 3) financialYearStartMarker--;
  const structuralYearEndMarker = (financialYearStartMarker + 1).toString().slice(-2);
  const localizedFiscalString = `${financialYearStartMarker.toString().slice(-2)}-${structuralYearEndMarker}`;
  
  const targetFormStringPrefix = `AVD/${sequencePrefix}/${localizedFiscalString}/`;
  const rawDataMatrix = sheet.getDataRange().getValues();
  let sequenceTrackerIndex = 0;

  for(let i = 1; i < rawDataMatrix.length; i++) {
     const identificationStringPointer = rawDataMatrix[i][0] ? rawDataMatrix[i][0].toString() : '';
     if(identificationStringPointer.startsWith(targetFormStringPrefix)) {
        const parsedSequenceInt = parseInt(identificationStringPointer.split('/').pop(), 10);
        if(!isNaN(parsedSequenceInt) && parsedSequenceInt > sequenceTrackerIndex) {
          sequenceTrackerIndex = parsedSequenceInt;
        }
     }
  }
  return `${targetFormStringPrefix}${(sequenceTrackerIndex + 1).toString().padStart(4, '0')}`;
}

/**
 * Commits security auditing trails to isolated log records for validation review history parsing.
 */
function logSystemAction(actorEmail, actionType, targetId) {
  try {
    const ss = SpreadsheetApp.openById(MASTER_DB_ID);
    const auditSheet = ss.getSheetByName('Audit_Logs');
    if (!auditSheet) return;

    const logId = Utilities.getUuid();
    const timestamp = new Date().toISOString();
    
    auditSheet.appendRow([
      logId,
      timestamp,
      actorEmail || 'SYSTEM_DAEMON',
      actionType,
      targetId || 'N/A'
    ]);
  } catch (criticalFailureObfuscation) {
    // Suppress logging engine loop failures to avoid locking runtime execution paths
    console.error("Audit Engine Failure: " + criticalFailureObfuscation.toString());
  }
}

/**
 * Legacyself-healing migration for scrambled column indices in the Parent_Tickets sheet.
 * Detects if existing rows were written in a scrambled column format and realigns them.
 */
function migrateParentTickets(ss) {
  try {
    // Realign sheet headers to standard naming conventions dynamically
    standardizeSpreadsheetHeaders(ss);
    
    const pSheet = ss.getSheetByName('Parent_Tickets');
    if (!pSheet) return;

    const range = pSheet.getDataRange();
    const values = range.getValues();
    if (values.length <= 1) return;

    const headers = values[0];
    
    // Find column indexes dynamically to ensure robust lookup
    const parentIdIdx = headers.indexOf('Parent_ID');
    const serviceReqIdx = headers.indexOf('Service_Request_ID');
    const companyIdx = headers.indexOf('Company_Name');
    const locationIdx = headers.indexOf('Location');
    const roomIdx = headers.indexOf('Room_Name');
    const reqByIdx = headers.indexOf('Requested_By');
    const salesOrderIdx = headers.indexOf('Sales_Order');
    const categoryIdx = headers.indexOf('Category');
    const issueTypeIdx = headers.indexOf('Issue_Type');
    const brandIdx = headers.indexOf('ProductMake') !== -1 ? headers.indexOf('ProductMake') : headers.indexOf('Brand');
    const modelIdx = headers.indexOf('ProductModel') !== -1 ? headers.indexOf('ProductModel') : headers.indexOf('Model');
    const serialIdx = headers.indexOf('ProductSerial') !== -1 ? headers.indexOf('ProductSerial') : headers.indexOf('Serial');
    const serviceTypeIdx = headers.indexOf('Service_Type');
    const descriptionIdx = headers.indexOf('Description');
    const openDateIdx = headers.indexOf('Open_Date');
    const closeDateIdx = headers.indexOf('Close_Date');
    const resolvedDaysIdx = headers.indexOf('Resolved_Days');
    const clientEmailIdx = headers.indexOf('Client_Email');
    const remarksIdx = headers.indexOf('Admin_Remarks');
    const statusIdx = headers.indexOf('Status');

    let modified = false;

    for (let i = 1; i < values.length; i++) {
      const row = values[i];
      
      const company = row[companyIdx];
      const room = row[roomIdx];
      const location = row[locationIdx];
      const category = row[categoryIdx];
      const issueType = row[issueTypeIdx];
      const brand = row[brandIdx];
      const model = row[modelIdx];
      const serial = row[serialIdx];
      const serviceType = row[serviceTypeIdx];
      const description = row[descriptionIdx];
      const openDate = row[openDateIdx];
      const clientEmail = row[clientEmailIdx];
      const remarks = row[remarksIdx];

      // A row is scrambled if the Company_Name column is empty, but Room_Name contains actual text data,
      // and Location is either empty or matches a Sales Order reference format.
      const looksScrambled = !company && room && (!location || location.toString().startsWith('SO') || /^\d+$/.test(location.toString()));

      if (looksScrambled) {
        const newRow = [...row];
        
        newRow[companyIdx] = room; // Company name was written to Room_Name
        newRow[locationIdx] = category; // Location was written to Category
        newRow[roomIdx] = row[reqByIdx]; // Room Name was written to Requested_By
        newRow[reqByIdx] = row[salesOrderIdx]; // Requester was written to Sales_Order
        newRow[salesOrderIdx] = location; // Sales Order was written to Location
        newRow[categoryIdx] = issueType; // Category was written to Issue_Type
        newRow[issueTypeIdx] = brand; // Issue Type was written to Brand
        newRow[brandIdx] = model; // Brand was written to Model
        newRow[modelIdx] = serial; // Model was written to Serial
        newRow[serialIdx] = serviceType; // Serial was written to Service_Type
        newRow[serviceTypeIdx] = ''; // Service Type was empty
        newRow[descriptionIdx] = description || openDate; // Description was written to Description or Open_Date
        
        // Correct date shift: if Open_Date column contains description text
        if (openDate && !(openDate instanceof Date) && isNaN(Date.parse(openDate))) {
          newRow[descriptionIdx] = openDate;
          const closeDateVal = row[closeDateIdx];
          if (closeDateVal && (closeDateVal instanceof Date || !isNaN(Date.parse(closeDateVal)))) {
            newRow[openDateIdx] = closeDateVal;
            newRow[closeDateIdx] = '';
          }
        }

        // Correct email shift: if Client_Email was empty but Remarks column contains the email address
        if (!clientEmail && remarks && remarks.toString().includes('@')) {
          newRow[clientEmailIdx] = remarks;
          newRow[remarksIdx] = '';
        }

        pSheet.getRange(i + 1, 1, 1, newRow.length).setValues([newRow]);
        modified = true;
      }
    }

    if (modified) {
      SpreadsheetApp.flush();
    }
  } catch (err) {
    console.error("Migration Utility Error: " + err.toString());
  }
}

/**
 * Automatically standardizes headers in existing spreadsheet sheets to resolve schema mismatches.
 * - Renames legacy Parent_Tickets headers: 'Brand' -> 'ProductMake', 'Model' -> 'ProductModel', 'Serial' -> 'ProductSerial'
 * - Renames legacy Request_Items headers: 'ProductBrand' -> 'ProductMake'
 */
function standardizeSpreadsheetHeaders(ss) {
  try {
    // 1. Realign Parent_Tickets headers
    const pSheet = ss.getSheetByName('Parent_Tickets');
    if (pSheet && pSheet.getLastRow() > 0) {
      const pHeadersRange = pSheet.getRange(1, 1, 1, pSheet.getLastColumn());
      const pHeaders = pHeadersRange.getValues()[0];
      let pModified = false;
      
      const brandIdx = pHeaders.indexOf('Brand');
      if (brandIdx !== -1) {
        pHeaders[brandIdx] = 'ProductMake';
        pModified = true;
      }
      
      const modelIdx = pHeaders.indexOf('Model');
      if (modelIdx !== -1) {
        pHeaders[modelIdx] = 'ProductModel';
        pModified = true;
      }
      
      const serialIdx = pHeaders.indexOf('Serial');
      if (serialIdx !== -1) {
        pHeaders[serialIdx] = 'ProductSerial';
        pModified = true;
      }
      
      if (pModified) {
        pHeadersRange.setValues([pHeaders]);
        Logger.log("Standardized Parent_Tickets header columns.");
      }
    }

    // 2. Realign Request_Items headers
    const itemSheet = ss.getSheetByName('Request_Items');
    if (itemSheet && itemSheet.getLastRow() > 0) {
      const iHeadersRange = itemSheet.getRange(1, 1, 1, itemSheet.getLastColumn());
      const iHeaders = iHeadersRange.getValues()[0];
      let iModified = false;
      
      const prodBrandIdx = iHeaders.indexOf('ProductBrand');
      if (prodBrandIdx !== -1) {
        iHeaders[prodBrandIdx] = 'ProductMake';
        iModified = true;
      }
      
      if (iModified) {
        iHeadersRange.setValues([iHeaders]);
        Logger.log("Standardized Request_Items header columns.");
      }
    }
  } catch (err) {
    Logger.log("Error standardizing spreadsheet headers: " + err.toString());
  }
}
